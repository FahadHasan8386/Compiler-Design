#include <iostream>
using namespace std;

bool isOperator(char ch)
{
    return (ch == '+' || ch == '-' || ch  == '*' || ch == '/' || ch == '=' || ch == '%');
}

int main ()
{
    string input ;
    cout << "Enter Expression : ";
    cin >> input;


    for( int i = 0 ; i < input[i] ;i++)
    {
        if(isOperator(input[i]))
        {
            cout << "Operator : " << input[i] << endl;
        }
    }
}
