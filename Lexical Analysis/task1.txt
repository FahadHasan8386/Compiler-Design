#include<iostream>
using namespace std ;

bool isNumberConstant(string str)
{
    if(str.empty())return false;

    for(int i = 0 ; i < str.size() ;i++)
    {

        if(str[i] < '0' || str[i] > '9')
        {
            return false;
        }
    }
    return true;
}

int main ()
{
    string input ;
    cout << "Enter Input : ";
    cin >> input;

    if(isNumberConstant(input))
    {
        cout << "Is a Num " << endl;
    }
    else
    {
        cout << "Is not a num " << endl;
    }
}
